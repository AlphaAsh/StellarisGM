sgm_add_hyperlane = {
	potential = {
		is_gm = yes
		exists = event_target:target_planet
		NOT = {
			solar_system = { has_hyperlane_to = event_target:target_planet.solar_system }  
		}
	}
	
	allow = {}

	effect = {
		solar_system = {
			add_hyperlane = {
				from = THIS
				to = event_target:target_planet.solar_system
			}
		}
	}
}
sgm_remove_hyperlane = {
	potential = {

			}
		}
	}
}

sgm_remove_all_hyperlanes = {
